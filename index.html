function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    let sumSheet = ss.getSheetByName("請求書サマリー")
      || ss.insertSheet("請求書サマリー");
    if (sumSheet.getLastRow() === 0) {
      sumSheet.appendRow(["ファイル名","請求書番号","請求日","入金期日",
        "発行元","宛先","小計","消費税","請求金額"]);
    }

    let detSheet = ss.getSheetByName("明細一覧")
      || ss.insertSheet("明細一覧");
    if (detSheet.getLastRow() === 0) {
      detSheet.appendRow(["請求書番号","請求日","取引日",
        "摘要","数量","単価","明細金額"]);
    }

    for (const inv of data.invoices) {
      sumSheet.appendRow([
        inv.filename, inv.invoice_number, inv.invoice_date,
        inv.due_date, inv.issuer, inv.recipient,
        inv.subtotal, inv.tax, inv.total
      ]);
      for (const item of (inv.items || [])) {
        detSheet.appendRow([
          inv.invoice_number, inv.invoice_date, item.date,
          item.description, item.quantity, item.unit_price, item.amount
        ]);
      }
    }

    return ContentService
      .createTextOutput(JSON.stringify({status:"ok"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:"error", message:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
