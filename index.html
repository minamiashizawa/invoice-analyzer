<script>

const GAS_URL = "https://script.google.com/macros/s/AKfycbz3Yr98fDFQwJuSwzoysNXTj1hJzcs-cg1e8hWkK6FSrEK90INgJ__r8TpGMwPG0fq1ag/exec";

const COMPANY_ID = "11966477";

const CLIENT_ID = "657848771310622";

const REDIRECT_URI = "https://minamiashizawa.github.io/invoice-analyzer/";

let files = [], invoices = [], freeeInvoices = [];

let freeeAccessToken = null;


function showTab(name) {

  document.querySelectorAll('.tab').forEach((t,i) => t.classList.toggle('active', ['freee','pdf'][i]===name));

  document.querySelectorAll('.tab-panel').forEach(p => p.classList.remove('active'));

  document.getElementById('tab-'+name).classList.add('active');

}


window.addEventListener('load', () => {

  const authUrl = `https://accounts.secure.freee.co.jp/public_api/authorize?client_id=${CLIENT_ID}&redirect_uri=${encodeURIComponent(REDIRECT_URI)}&response_type=token&prompt=select_company`;

  document.getElementById('freeeLoginBtn').href = authUrl;


  // URLのhashからトークン取得

  const hash = window.location.hash;

  if (hash.includes('access_token')) {

    const params = new URLSearchParams(hash.replace('#',''));

    freeeAccessToken = params.get('access_token');

    sessionStorage.setItem('freeeToken', freeeAccessToken);

    document.getElementById('freeeLoginBtn').style.display = 'none';

    document.getElementById('freeeLoggedIn').style.display = 'block';

    window.history.replaceState({}, document.title, window.location.pathname);

  } else {

    const saved = sessionStorage.getItem('freeeToken');

    if (saved) {

      freeeAccessToken = saved;

      document.getElementById('freeeLoginBtn').style.display = 'none';

      document.getElementById('freeeLoggedIn').style.display = 'block';

    } else {

      document.getElementById('freeeLoggedIn').style.display = 'none';

    }

  }

});


async function fetchFreee() {

  if (!freeeAccessToken) {

    setStatus('freeeStatus','先にFreeeでログインしてください','err');

    return;

  }

  const start = document.getElementById('startDate').value.trim();

  const end = document.getElementById('endDate').value.trim();

  setStatus('freeeStatus','Freeeから取得中...','info');

  try {

    let url = `https://api.freee.co.jp/api/1/invoices?company_id=${COMPANY_ID}&limit=100`;

    if (start) url += `&issue_date_start=${start}`;

    if (end) url += `&issue_date_end=${end}`;


    const r = await fetch(url, {

      headers: { 'Authorization': `Bearer ${freeeAccessToken}`, 'Content-Type': 'application/json' }

    });

    const d = await r.json();

    if (d.errors || !d.invoices) throw new Error(JSON.stringify(d.errors || d));


    freeeInvoices = d.invoices;

    setStatus('freeeStatus', `✅ ${freeeInvoices.length}件の請求書を取得しました`, 'ok');

    renderFreeeList();

  } catch(e) {

    setStatus('freeeStatus', `❌ エラー: ${e.message}`, 'err');

  }

}


function renderFreeeList() {

  document.getElementById('freeeList').innerHTML = freeeInvoices.map(inv => `<tr>

    <td class="b">${inv.invoice_number||'—'}</td>

    <td>${inv.issue_date||'—'}</td>

    <td>${inv.partner_name||'—'}</td>

    <td class="r">${fmt(inv.total_amount||0)}</td>

    <td>${inv.invoice_status||'—'}</td>

  </tr>`).join('');

  document.getElementById('freeeResultCard').style.display = 'block';

}


async function sendFreeeToSheets() {

  setStatus('freeeSheetStatus','Google Sheetsへ書き込み中...','info');

  try {

    const invs = freeeInvoices.map(inv => ({

      filename: `Freee_${inv.invoice_number||inv.id}`,

      invoice_number: inv.invoice_number||'',

      invoice_date: inv.issue_date||'',

      due_date: inv.payment_date||'',

      issuer: 'eLife株式会社',

      recipient: inv.partner_name||'',

      subtotal: inv.sub_total||0,

      tax: inv.tax_amount||0,

      total: inv.total_amount||0,

      items: (inv.invoice_lines||[]).map(line => ({

        date: inv.issue_date||'',

        description: line.description||'',

        quantity: line.quantity||0,

}

function addFiles(fs) {

  renderChips();

}

function removeFile(i) { files.splice(i,1); renderChips(); }

function renderChips() {

  document.getElementById('chips').innerHTML = files.map((f,i) =>

  document.getElementById('runBtn').disabled = files.length === 0;

}


function setStatus(id, msg, type) {

  el.textContent = msg; el.className = 'status '+type;

}


async function toB64(file) {

  return new Promise((res,rej) => {

    const r = new FileReader();

    r.onload = () => res(r.result.split(',')[1]);

    r.onerror = rej;

    r.readAsDataURL(file);

  });

}


async function extract(b64, name, apiKey) {

  const resp = await fetch('https://api.anthropic.com/v1/messages', {

    method:'POST',

    headers:{

      'Content-Type':'application/json',

      'x-api-key': apiKey,

      'anthropic-version': '2023-06-01',

      'anthropic-dangerous-direct-browser-access': 'true'

    },

    body: JSON.stringify({

      model:'claude-sonnet-4-20250514', max_tokens:1000,

      messages:[{role:'user', content:[

        {type:'document', source:{type:'base64', media_type:'application/pdf', data:b64}},

        {type:'text', text:`請求書を解析してJSONのみ返してください（説明不要）:

{"invoice_number":"","invoice_date":"YYYY-MM-DD","due_date":"YYYY-MM-DD","issuer":"","recipient":"","subtotal":0,"tax":0,"total":0,"items":[{"date":"","description":"","quantity":0,"unit_price":0,"amount":0}]}`}

      ]}]

    })

  });

  const d = await resp.json();

  if (d.error) throw new Error(d.error.message);

  const txt = d.content.map(c=>c.text||'').join('').replace(/```json|```/g,'').trim();

  const p = JSON.parse(txt);

  p.filename = name;

  return p;

}


async function runAll() {

  const apiKey = document.getElementById('apiKey').value.trim();

  if (!apiKey) { setStatus('mainStatus','APIキーを入力してください','err'); return; }

  document.getElementById('loading').style.display = 'block';

  document.getElementById('results').style.display = 'none';

  document.getElementById('mainStatus').className = 'status';

  invoices = [];

  try {

    for (let i=0; i<files.length; i++) {

      const pct = Math.round((i/files.length)*70);

      document.getElementById('progressFill').style.width = pct+'%';

      document.getElementById('loadingMsg').textContent = `AIで解析中 (${i+1}/${files.length}): ${files[i].name}`;

      invoices.push(await extract(await toB64(files[i]), files[i].name, apiKey));

    }

    document.getElementById('loadingMsg').textContent = 'Google Sheetsへ書き込み中...';

    document.getElementById('progressFill').style.width = '90%';

    await fetch(GAS_URL, { method:'POST', mode:'no-cors', headers:{'Content-Type':'application/json'}, body:JSON.stringify({invoices}) });

    document.getElementById('progressFill').style.width = '100%';

    document.getElementById('loading').style.display = 'none';

    renderResults();

    setStatus('mainStatus', `✅ ${invoices.length}件の請求書をGoogle Sheetsへ書き込みました！`, 'ok');

  } catch(e) {

    document.getElementById('loading').style.display = 'none';

    setStatus('mainStatus', `❌ エラー: ${e.message}`, 'err');

  }

}


function fmt(n) { return Number(n).toLocaleString('ja-JP')+'円'; }


function renderResults() {

  const total = invoices.reduce((s,v)=>s+(v.total||0),0);

  const tax = invoices.reduce((s,v)=>s+(v.tax||0),0);

  const items = invoices.reduce((s,v)=>s+(v.items?.length||0),0);

  document.getElementById('statGrid').innerHTML = `

    <div class="stat"><div class="l">請求書件数</div><div class="v">${invoices.length}</div><div class="s">件</div></div>

    <div class="stat"><div class="l">請求金額合計</div><div class="v" style="font-size:1rem">${fmt(total)}</div></div>

    <div class="stat"><div class="l">消費税合計</div><div class="v" style="font-size:1rem">${fmt(tax)}</div></div>

    <div class="stat"><div class="l">明細件数合計</div><div class="v">${items}</div><div class="s">件</div></div>

  `;

  document.getElementById('sumTbody').innerHTML = invoices.map(inv=>`<tr>

    <td>${inv.filename}</td><td class="b">${inv.invoice_number||'—'}</td>

    <td>${inv.invoice_date||'—'}</td><td>${inv.due_date||'—'}</td>

    <td>${inv.issuer||'—'}</td>
